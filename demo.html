modernAlert v1.00 Examples

#1 Basic Example

// on the click event of the button
modernAlert('Ask a question to the user?');

// this is the basic buttons layout , and the full version of it will be:
modernAlert('Ask a question to the user?', {buttonsLayout: 'ok'}, myCallBackFunction);

#2 Yes/No Example

(Check the browser's console to see the result!)

// on the click event of the button
modernAlert('Ask a question to the user?', {buttonsLayout: 'yesNo'}, result => {
  console.log(result); // true or false
});

#3 Custom buttons Example

(Check the browser's console to see the result!)

// define myAlert function
function myAlert(){
  modernAlert('Ask a question to the user?', {
    buttonsLayout: 'custom',
    buttonsStructure: [
      {label: 'button1', return: 'return something #1'},
      {label: 'button2', return: 'return something #2'},
      {label: 'button3', return: 'return something #3'},
    ]
  },
  result => {
    console.log(result); // 'return something #1' , 'return something #2' or 'return something #3'
  });
}

// now you can call myAlert() , on your button's click event (e.g onclick="myAlert();")
